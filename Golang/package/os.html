//创建目录
os.Mkdir("abc", os.ModePerm);

//创建多级目录
os.MkdirAll("abc/d/e/f", os.ModePerm);

//删除文件或目录
os.Remove("abc/d/e/f");

//删除指定目录下所有文件
os.RemoveAll("abc");

//重命名文件
os.Rename("./2.txt", "./2_new.txt");

//判断是否为同一文件
//unix下通过底层结构的设备和索引节点是否相同来判断
//其他系统可能是通过文件绝对路径来判断
fs1, _ := f1.Stat();
f2, _ := os.Open("./1.txt");
fs2, _ := f2.Stat();
fmt.Println(os.SameFile(fs1, fs2));

//返回临时目录
fmt.Println(os.TempDir());
